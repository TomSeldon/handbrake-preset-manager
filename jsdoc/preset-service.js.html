'use strict';

goog.provide('hpm.repository.preset.Service');

/**
 * Preset Repository.
 *
 * @param {hpm.datacontext.Service} DataContext
 * @constructor
 */
var PresetRepository = function(DataContext)
{
    /**
     * @type {hpm.datacontext.Service}
     * @private
     */
    this.dataContext_ = DataContext;
};

/**
 * Create generic Presets query.
 *
 * @return {breeze.EntityQuery}
 */
PresetRepository.prototype.createPresetsQuery = function()
{
    return this.dataContext_.createQuery().from('Presets');
};

/**
 * Get all presets.
 *
 * @return {*|Promise}
 */
PresetRepository.prototype.getAllPresets = function()
{
    return this.dataContext_.executeQuery(
        this.createPresetsQuery()
    );
};

/**
 * Get presets using the provided query.
 *
 * TODO: Should this accept a query or a Predicate?
 *
 * @param {breeze.EntityQuery} query
 * @return {*|Promise}
 */
PresetRepository.prototype.getPresets = function(query)
{
    return this.dataContext_.executeQuery(query);
};

/**
 * Create a new Preset.
 *
 * @return {*|Entity}
 */
PresetRepository.prototype.createPreset = function()
{
    return this.dataContext_.createEntity('Preset');
};

/**
 * @type {PresetRepository}
 */
hpm.repository.preset.Service = PresetRepository;
