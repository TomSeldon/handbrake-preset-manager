'use strict';

goog.provide('hpm.repository.category.Service');

/**
 * Category Repository.
 *
 * @param {hpm.datacontext.Service} DataContext
 * @constructor
 */
var CategoryRepository = function(DataContext)
{
    /**
     * @type {hpm.datacontext.Service}
     * @private
     */
    this.dataContext_ = DataContext;
};

/**
 * Create generic Categories query.
 *
 * @return {breeze.EntityQuery}
 * @private
 */
CategoryRepository.prototype.createCategoriesQuery_ = function()
{
    return this.dataContext_.createQuery().from('Categories');
};

/**
 * Get all categories.
 *
 * @return {*|Promise}
 */
CategoryRepository.prototype.getAllCategories = function()
{
    return this.dataContext_.executeQuery(
        this.createCategoriesQuery_()
    );
};

/**
 * Get categories using the provided query.
 *
 * TODO: Should this accept a query or a Predicate?
 *
 * @param {breeze.EntityQuery} query
 * @return {*|Promise}
 */
CategoryRepository.prototype.getCategories = function(query)
{
    return this.dataContext_.executeQuery(query);
};

/**
 * Create a new Category.
 *
 * @return {*|Entity}
 */
CategoryRepository.prototype.createCategory = function()
{
    return this.dataContext_.createEntity('Category');
};

/**
 * @type {CategoryRepository}
 */
hpm.repository.category.Service = CategoryRepository;
