'use strict';

goog.provide('hpm.breeze.entityquery.Service');

/**
 * @param {breeze.EntityQuery} EntityQuery
 * @constructor
 */
var EntityQueryFactory = function(EntityQuery)
{
    /**
     * @type {breeze.EntityQuery}
     * @private
     */
    this.EntityQuery_ = EntityQuery;
};

/**
 * @return {breeze.EntityQuery}
 */
EntityQueryFactory.prototype.createQuery = function()
{
    return new this.EntityQuery_();
};

/**
 * @type {EntityQueryFactory}
 */
hpm.breeze.entityquery.Service = EntityQueryFactory;
