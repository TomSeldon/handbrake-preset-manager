'use strict';

goog.provide('hpm.datacontext.Service');

/**
 * @param {hpm.breeze.entityquery.Factory} EntityQueryFactory
 * @param {hpm.breeze.entitymanager.Service} entityManagerFactory
 * @constructor
 * @ngInject
 */
var DataContext = function(EntityQueryFactory, entityManagerFactory)
{
    /**
     * @private
     * @type {breeze}
     */
    this.entityQueryFactory_ = EntityQueryFactory;

    /**
     * Get the shared preset manager.
     *
     * @private
     * @type {breeze.EntityManager}
     */
    this.entityManager_ = entityManagerFactory.getSharedManager();

    /**
     * To be populated with lookup data on
     * initialization.
     *
     * @private
     * @type {{}}
     */
    this.lookups_ = {};

    /**
     * Keep track of whether this data context
     * has been initialized.
     *
     * @private
     * @type {boolean}
     */
    this.initialized_ = false;

    /**
     * Declare API.
     *
     * @private
     */
    this.api_ = {
        initialize: this.initialize,
        createQuery: this.createQuery,
        createEntity: this.createEntity,
        getLookups: this.getLookups,
        cancelAllChanges: this.cancelAllChanges,
        saveChanges: this.saveChanges,
        hasChanges: this.hasChanges,
        hasFetchedMetadata: this.hasFetchedMetadata
    };

    /**
     * Exposes the above declared API
     * after first binding the returned methods
     * to the correct context.
     *
     * @return {Object}
     */
    return this.exposeApi_();
};

/**
 * Returns the API after binding the methods
 * to the correct context.
 *
 * @private
 * @return {Object}
 */
DataContext.prototype.exposeApi_ = function()
{
    for (var methodName in this.api_) {
        if (this.api_.hasOwnProperty(methodName)) {
            this.api_[methodName] = this.api_[methodName].bind(this);
        }
    }

    return this.api_;
};

/**
 * @throws {Error}
 * @return {Promise}
 */
DataContext.prototype.initialize = function()
{
    var lookupsQuery;

    if (this.initialized_ === true) {
        throw new Error(
            'Data context initialization attempted multiple times.'
        );
    }

    // Get lookups
    lookupsQuery = this.createQuery().from('Lookups');

    return this.entityManager_.executeQuery(lookupsQuery)
        .then(function(data) {
            this.lookups_ = data.results[0];
            this.initialized_ = true;
        }.bind(this));
};

/**
 * @return {breeze.EntityQuery}
 */
DataContext.prototype.createQuery = function()
{
    return new this.entityQueryFactory_.createQuery();
};

/**
 * Factory method for creating a new entity of the specified type.
 *
 * @param {string} entityTypeName
 * @return {*|Entity}
 */
DataContext.prototype.createEntity = function(entityTypeName)
{
    return this.entityManager_.createEntity(entityTypeName);
};

/**
 * @param {breeze.EntityQuery} query
 * @return {Promise}
 */
DataContext.prototype.executeQuery = function(query)
{
    return this.entityManager_.executeQuery(query);
};

/**
 * @return {Object}
 */
DataContext.prototype.getLookups = function()
{
    return this.lookups_;
};

/**
 * Save changes in the entity manager.
 *
 * @return {*|Promise}
 */
DataContext.prototype.saveChanges = function()
{
    return this.entityManager_.saveChanges();
};

/**
 * Cancel any changes present in the entity manager.
 *
 * @return {*}
 */
DataContext.prototype.cancelAllChanges = function()
{
    return this.entityManager_.rejectChanges();
};

/**
 * Checks entity manager for changes.
 *
 * @return {Boolean}
 */
DataContext.prototype.hasChanges = function()
{
    return this.entityManager_.hasChanges();
};

/**
 * Check whether metadata has been returned for the data service.
 *
 * @return {Boolean}
 */
DataContext.prototype.hasFetchedMetadata = function()
{
    var serviceName = this.entityManager_.serviceName,
        metadataStore = this.entityManager_.metadataStore;

    return metadataStore.hasMetadataFor(serviceName);
};

/**
 * @type {DataContext}
 */
hpm.datacontext.Service = DataContext;
